#version 460 core

uniform sampler2D diffuseTexture;
uniform vec3 DIFFUSE;
uniform vec3 SPECULAR;
uniform float shininess;

in vec3 fragNormal;
in vec3 fragPosition;
in vec2 fragTexCoords;
in vec3 fragWorldPosition;

layout (location = 0) out vec3 gPosition;
layout (location = 1) out vec3 gNormal;
layout (location = 2) out vec4 gAlbedoSpec;

void main() {
    gPosition = fragWorldPosition;
    gNormal = normalize(fragNormal);
    
    vec4 texColor = texture(diffuseTexture, fragTexCoords);
    vec3 albedo = DIFFUSE * texColor.rgb;
    float specularStrength = (SPECULAR.x + SPECULAR.y + SPECULAR.z) / 3.0;
    
    gAlbedoSpec = vec4(albedo, specularStrength);
}

#version 460 core

in vec2 fragTexCoords;

uniform sampler2D gPosition;
uniform sampler2D gNormal;
uniform sampler2D gAlbedoSpec;

uniform vec3 viewPos;
uniform float shininess;
uniform float far_plane;

uniform int useAmbient;
uniform vec3 ambientColor;
uniform float ambientIntensity;

uniform int numDirLights;
uniform int numSpotLights;
uniform int numPointLights;

struct DirLight {
    vec3 direction;
    vec3 color;
    float intensity;
};

struct SpotLight {
    vec3 position;
    vec3 direction;
    vec3 color;
    float intensity;
    float cutOff;
    float outerCutOff;
    float constant;
    float linear;
    float quadratic;
};

struct PointLight {
    vec3 position;
    vec3 color;
    float intensity;
    float constant;
    float linear;
    float quadratic;
};

uniform DirLight dirLights[4];
uniform SpotLight spotLights[8];
uniform PointLight pointLights[16];

out vec4 fragColor;

void main() {
    vec3 fragPos = texture(gPosition, fragTexCoords).rgb;
    vec3 normal = texture(gNormal, fragTexCoords).rgb;
    vec4 albedoSpec = texture(gAlbedoSpec, fragTexCoords);
    vec3 albedo = albedoSpec.rgb;
    float specular = albedoSpec.a;

    if (length(fragPos) == 0.0 && length(normal) == 0.0) {
        fragColor = vec4(0.0, 0.0, 0.0, 1.0);
        return;
    }

    vec3 viewDir = normalize(viewPos - fragPos);
    vec3 result = vec3(0.0);

    if (useAmbient == 1) {
        result += ambientColor * ambientIntensity * albedo;
    }

    for (int i = 0; i < numDirLights; i++) {
        vec3 lightDir = normalize(-dirLights[i].direction);
        float diff = max(dot(normal, lightDir), 0.0);
        vec3 reflectDir = reflect(-lightDir, normal);
        float spec = pow(max(dot(viewDir, reflectDir), 0.0), shininess);
        vec3 diffuse = diff * albedo * dirLights[i].color * dirLights[i].intensity;
        vec3 specularVec = spec * specular * dirLights[i].color * dirLights[i].intensity;
        result += diffuse + specularVec;
    }

    for (int i = 0; i < numSpotLights; i++) {
        vec3 lightDir = normalize(spotLights[i].position - fragPos);
        float distance = length(spotLights[i].position - fragPos);
        float attenuation = 1.0 / (spotLights[i].constant + spotLights[i].linear * distance + spotLights[i].quadratic * (distance * distance));

        float theta = dot(lightDir, normalize(-spotLights[i].direction));
        float epsilon = spotLights[i].cutOff - spotLights[i].outerCutOff;
        float intensity = clamp((theta - spotLights[i].outerCutOff) / epsilon, 0.0, 1.0);

        float diff = max(dot(normal, lightDir), 0.0);
        vec3 reflectDir = reflect(-lightDir, normal);
        float spec = pow(max(dot(viewDir, reflectDir), 0.0), shininess);

        vec3 diffuse = diff * albedo * spotLights[i].color * spotLights[i].intensity * attenuation * intensity;
        vec3 specularVec = spec * specular * spotLights[i].color * spotLights[i].intensity * attenuation * intensity;
        result += diffuse + specularVec;
    }

    for (int i = 0; i < numPointLights; i++) {
        vec3 lightDir = normalize(pointLights[i].position - fragPos);
        float distance = length(pointLights[i].position - fragPos);
        float attenuation = 1.0 / (pointLights[i].constant + pointLights[i].linear * distance + pointLights[i].quadratic * (distance * distance));

        float diff = max(dot(normal, lightDir), 0.0);
        vec3 reflectDir = reflect(-lightDir, normal);
        float spec = pow(max(dot(viewDir, reflectDir), 0.0), shininess);

        vec3 diffuse = diff * albedo * pointLights[i].color * pointLights[i].intensity * attenuation;
        vec3 specularVec = spec * specular * pointLights[i].color * pointLights[i].intensity * attenuation;
        result += diffuse + specularVec;
    }

    fragColor = vec4(result, 1.0);
}

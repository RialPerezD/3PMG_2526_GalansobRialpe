#version 460 core
uniform mat4 MVP;
uniform vec3 DIFFUSE;
uniform vec3 SPECULAR;
uniform vec3 AMBIENT;
in vec3 position;
in vec3 normal;
out vec3 fragNormal;
out vec3 fragPosition;
out vec3 fragColor;

void main() {
    gl_Position = MVP * vec4(position, 1.0);
    fragNormal = normalize(normal);
    fragPosition = position;
}

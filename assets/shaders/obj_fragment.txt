#version 460 core
uniform vec3 DIFFUSE;
uniform vec3 SPECULAR;
uniform vec3 AMBIENT;

in vec3 fragNormal;
in vec3 fragPosition;
out vec4 fragColor;

void main() {
    // Luz direccional (fija, hacia abajo en el eje Y)
    vec3 lightDir = normalize(vec3(0.0, -1.0, 0.0));  // Direccion de la luz

    // C치mara fija (en el eje Z)
    vec3 viewDir = normalize(vec3(0.0, 0.0, 5.0) - fragPosition);  // Direcci칩n de la camara
    vec3 reflectDir = reflect(-lightDir, fragNormal);  // Direccion de la reflexi칩n de la luz

    // Componente ambiental (constantemente iluminado por una luz ambiental)
    vec3 ambient = AMBIENT;

    // Componente difusa (Lambertian)
    float diff = max(dot(fragNormal, lightDir), 0.0);
    vec3 diffuse = DIFFUSE * diff;

    // Componente especular (Phong)
    float spec = pow(max(dot(viewDir, reflectDir), 0.0), 32);  // Exponente: 32
    vec3 specular = SPECULAR * spec;

    // Combinar los componentes de iluminaci칩n
    vec3 color = ambient + diffuse + specular;

    // El color final del fragmento
    fragColor = vec4(fragPosition, 1.0);
}

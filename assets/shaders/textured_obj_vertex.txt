#version 460 core

uniform mat4 MVP;
uniform mat4 model;

in vec3 position;
in vec3 normal;
in vec2 uv;

out vec3 fragNormal;
out vec3 fragPosition;
out vec2 fragTexCoords;

void main() {
    gl_Position = MVP * vec4(position, 1.0);

    fragPosition = vec3(model * vec4(position, 1.0));
    fragNormal = normalize(mat3(model) * normal);
    fragTexCoords = uv;
}
